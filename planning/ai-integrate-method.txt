This allows your hamburger menu and your Gemini code to run smoothly together in the browser.javascript// 1. Import the browser-compatible Google Gen AI SDK
import { GoogleGenAI } from 'https://esm.sh';

// 2. Your existing frontend code will now work perfectly
var hamburger = document.getElementById('hamburger');
hamburger.addEventListener('click', () => {
    console.log("Hamburger menu clicked!");
});

// 3. Initialize Gemini with your API Key
// Hardcode your key ONLY for quick local testing. Do not push this to GitHub!
const ai = new GoogleGenAI({ apiKey: "YOUR_GEMINI_API_KEY_HERE" });

async function askGemini() {
  try {
    const outputElement = document.getElementById('ai-output');
    outputElement.innerText = "Thinking...";

    const response = await ai.models.generateContent({
      model: 'gemini-2.5-flash',
      contents: 'Give me a quick 1-sentence tip for studying coding.',
    });

    outputElement.innerText = response.text;
  } catch (error) {
    console.error("Gemini Error:", error);
  }
}

// Attach the function to a frontend button click
document.getElementById('ai-btn').addEventListener('click', askGemini);
